111  (

115  );
116 
117  m_right = enVector3
118  (
119  std::sinf(m_horizantalAngle - (glm::pi<float>() / 2)),
120  0.0f,
121  std::cosf(m_horizantalAngle - (glm::pi<float>() / 2))
122  );
123 
124  m_up = glm::normalize(glm::cross(m_front, m_right));
125 
126  m_projection = glm::perspectiveFovLH(m_fov, m_width, m_height, m_near, m_far);
127 
128  m_view = glm::lookAtLH(m_position, m_position + m_front, m_up);
129 
130  glm::mat4x4 zRotation(1.0f);
131  const enVector4 FirstRow(std::cosf(m_zAngle), -std::sinf(m_zAngle), 0.0f, 0.0f);

133

118 float
119 imGuiManager::calculateAverageFPS(float deltaTime)
120 {
121  // gets the fps
122  float const fps = 1.0f / deltaTime;
123  // contains the results for later to average them out
124  static float fpsTimes[s_fpsSamplesCount] = { 0 };
125  static uint32_t currenFpsIndex = 0;
126 
127  if (currenFpsIndex > s_fpsSamplesCount - 1)
128  { currenFpsIndex = 0; }
129 
130  fpsTimes[currenFpsIndex] = fps;
131  currenFpsIndex++;
132 
133  float averageFps = 0.0f;
134  for (size_t i = 0; i < s_fpsSamplesCount; ++i)
135  {
136  averageFps += fpsTimes[i];
137  }
138  return averageFps = averageFps / s_fpsSamplesCount;
139 }
